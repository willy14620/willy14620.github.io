## 我的程式碼
```c
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int main(){
	int choice = 0;
	char str[9999];
	while(fgets(str,sizeof(str),stdin)){
		for(int i=0;i<strlen(str)-1;i++){
			if(str[i] == '"')
				(choice == 0) ? (printf("``"),choice = 1) : (printf("''"),choice = 0);
			else
				printf("%c",str[i]);
		}
		printf("\n");		
	}
	return 0;
}
```
